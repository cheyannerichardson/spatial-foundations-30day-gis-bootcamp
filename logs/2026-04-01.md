### 1. `validate_dataset` function

```python
def validate_dataset(df, analyst="Cheyanne Richardson"):
    """
    Valid forensic‑style validation check for gravesoil data.
    Logs schema, duplicates, ranges, and domain plausibility.
    Returns: status ("PASS", "REVIEW REQUIRED") and flags list.
    """
    print("\nSTEP 2: VALIDATION CHECKS")
    print("=" * 60)

    print("\n1) DATA TYPES:")
    print(df.dtypes)

    dup_count = df.duplicated().sum()
    print("\n2) DUPLICATE ROWS:")
    print(f"Duplicate rows: {dup_count}")

    print("\n3) RANGE CHECKS:")
    for col in df.columns:
        if pd.api.types.is_numeric_dtype(df[col]):
            print(
                f"{col}: min={df[col].min():.4f}, "
                f"max={df[col].max():.4f}, mean={df[col].mean():.4f}"
            )

    print("\n4) PLAUSIBILITY FLAGS:")
    flags = []
    if (df.select_dtypes(include='number') < 0).any().any():
        flags.append("Negative numeric values detected")
    if 'pH' in df.columns and ((df['pH'] < 0) | (df['pH'] > 14)).any():
        flags.append("pH out of valid range 0–14")

    if dup_count == 0 and not flags:
        print("✅ VALIDATION PASS")
    else:
        print("⚠️ VALIDATION REVIEW REQUIRED")

    audit_status = "PASS"
    if dup_count > 0 or flags:
        audit_status = "REVIEW REQUIRED"

    print(f"\nVALIDATION STATUS: {audit_status}")
    print(f"Analyst: {analyst}")
    print("-" * 60)

    return audit_status, flags
```
